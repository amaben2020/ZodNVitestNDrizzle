 iLike ensures that the database query is case insensitive

 const results = await db.query.users
 .findMany({
 limit,
 offset,
 where: or(
 ilike(users.jobTitle, q),
 ilike(users.name, q),
 ilike(users.firstName, q),
 ilike(users.lastName, q),
 ilike(users.email, q)
 ),
 })
 .execute();
